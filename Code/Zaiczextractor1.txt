# Function to convert text to a data frame
create_df = function ( text   )  {
  a0 = gsub("\\[.+?\\]", "", text)
  a1 = gsub( "\n" , " ",  a0   )
  a2 = gsub( ":", ","  ,  a1)
  a3 =  unlist ( strsplit(a2,",") )
  a4 = a3[-1] 
  a4 = str_trim(a4)
  
  output = data.frame( word = a4,
                      time = deparse(substitute(text)),
                      stringsAsFactors =F )  
   }
